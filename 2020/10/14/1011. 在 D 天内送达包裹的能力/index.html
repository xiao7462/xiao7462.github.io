    class Solution {
    public int shipWithinDays(int[] weights, int D) {
        int l = 0;
        int r = 0;
        for(int i =0; i < weights.length; i++){
            l = Math.max(l, weights[i]);
            r+= weights[i];
        }
        while( l < r){
            int mid = l + (r-l)/2;
            if( check(weights, D, mid)){
                r = mid;
            }else{
                l = mid+1;
            }
        }
        return l;

    }

    private boolean check(int[] weights, int D, int mid){
        int temp = mid;
        for(int i : weights){
            if(i>temp){
                temp = mid;
                D--;
            }
            temp-=i;
        }
        return D>0;
    }
}
