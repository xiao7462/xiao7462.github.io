    class Solution {
    public int removeDuplicates(int[] nums) {
       //[0,0,1,1,1,2,2,3,3,4]
       // p   q 
       int p = 0;
       int q = 1;
       while( q < nums.length){
           if( nums[q] > nums[p]){
               nums[p+1] = nums[q];
               p++;
           }
           q++;
       } 
       return p+1;
    }
}
